<root> ::= <remove-movie> | <add-movie> | <print-movies> | <compound-query>
<remove-movies> ::= remove-movies <movies>
<add-movies> ::= add-movies <movies>
<print-movies> ::= print-movies <movies>
<compound-query> ::= <root> " & " <root>

<movies> ::= <director><title><year><id>
<director> ::= <string>
<title> ::= <string> | <string><title>
<string> ::= <char> | <char><string>
<year> ::= <integer>
